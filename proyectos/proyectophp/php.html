class Cafetera{
private $capacidadMax;
private $cantActual;

public function __construct($cM,$cA){
$this->capacidadMax=$cM;
$this->cantActual=$cA;
}

public function getCapaMax(){
return $this->capacidadMax;
}

public function getCantAct(){
return $this->cantActual;
}

public function setCapaMax($ncm){
$this->capacidadMax = $ncm;
}

public function setCantAct($nca){
$this->cantActual = $nca;
}

public function __toString(){
return "La cantidad actual de la cafetera es ".$this->getCantAct()."\n";
}

public function llenarCafete(){
$this->setCantAct($this->getCapaMax());
}

public function servirTaza($taza){
if(($this->getCantAct()-$taza)) {
echo "No se puede, falta cafe \n";
}else{
$this->setCantAct($this->getCantAct()-$taza);
if($this->getCantAct()$taza){
echo "Su taza no va completamente llena... \n";
}
}
}
public function vaciarCafete(){
$this->setCantAct(0);
}

public function agregarCafe($c){
if(($this->getCantAct()+$c)>$this->getCapaMax()){
echo "Esa cantidad sobrepasa el limite...\n";
}else{
$this->setCantAct($this->getCantAct()+$c);
}
}
}
